Get the maximum and minimum of GDP per capita for all continents.

gapmnd_gdp <-gapminder |> 
  select(year, continent,gdpPercap)|>
  group_by(year,continent)|>
  summarise_if(is.numeric, funs(min,max),na.rm = TRUE)
